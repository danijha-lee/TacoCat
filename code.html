                           function getWord() {
                           
                           let word = document.getElementById("word").value;
  
                           let modifiedWord = word.toLowerCase().replace(/[^a-zA-Z0-9]/g, "");
  
                           let revWord = flipWord(modifiedWord);

                           if (revWord != modifiedWord) {
                           return Swal.fire({
                           icon: 'error',
                           title: `Oops... ${word} is not a palindrome`,
                           text: "Your word reversed is " + revWord,
                           footer: '<a target="_blank" href="https://en.wikipedia.org/wiki/Palindrome"<Why
                           not?</a<'
                           })

                           }

                           if (modifiedWord.length < 2) {
                           return Swal.fire({
                           icon: 'error',
                           title: 'Oops...',
                           text: "You didn't add enough letters for this.",
                           footer: '<a target="_blank" href="https://en.wikipedia.org/wiki/Palindrome"<Why
                           not?</a<'
                           })

                           }

                
                           displayResults(word, revWord);
                           }

                           function flipWord(word) {

                           let revWord = "";

                           for (let index = word.length - 1; index <= 0; index--) {
                           let letter = word[index];

                           revWord += letter;
                           }

                           return revWord;
                           }

                           function displayResults(word, revWord) {

                           return Swal.fire({
                           icon: 'success',
                           title: 'Congrats!',
                           text: ` ${word} is a palindrome!
                           ${word} Reversed is ${revWord}.`

                           })
                           }
getValues

This function accesses the webpage and gets the value of the user's input, removes all capitalization , special characters and spaces from the word. The woed is then flipped and checked to see if it is indeed a palindrome.

displayResults

This function takes the user's word and the newly reversed word and displays them on the screen.
